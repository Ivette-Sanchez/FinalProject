Min-Max Normalization Program

This program, created for a Data Mining coursework assignment, normalizes employee salary data using two statistical methods: Min-Max Normalization and Z-Score Normalization. It prompts the user for employee names and salaries, calculates normalization metrics, and displays standardized values for each employee.

Technolgy used: C++ 


                        
                        
            //Ivette Sanchez 
            //DataMining Min-Max Normalization Program
            
            #include <iostream>
            #include <cmath>
            #include <vector>
            using namespace std;

            int main() {
                int numEmployees;
                cout << "This is Ivettes Min-Max Normalization Program." << endl;
                cout << "Enter the number of employees: ";
                cin >> numEmployees;

                vector<int> salary(numEmployees); // create a vector the size of the number of employees to store salary.
                vector<string> employeeName(numEmployees); // create vector size of number of employees to store employee names.

                cout<<"Enter employee name and salaries: "<<endl;

                for (int i = 0; i < numEmployees; i++) { // loop through each employee to add theur name and salary.
                    cout << "Employee name: ";
                    cin >> employeeName[i];
                    cout << "Employee Salary: ";
                    cin >> salary[i];
                }

                double minSalary = salary[0]; // intialize both min/max salary to the first salary
                double maxSalary = salary[0]; // ^

                // Finding the minimum and maximum of salary
                for (int i = 0; i < numEmployees; i++) {
                    if (salary[i] < minSalary) // update minsalary if smaller salary is found
                        minSalary = salary[i];

                    if (salary[i] > maxSalary) // update maxsalary if smaller salary is found
                        maxSalary = salary[i];
                }

                // MIN-MAX NORMALIZATION
                cout << "Min-Max Normalization" << endl;

                for (int i = 0; i < numEmployees; i++) { // loop through each employee and calc the minmaxnorm for each
                    double norm = (salary[i] - minSalary) / (maxSalary - minSalary);
                    cout << "Employee " << employeeName[i] << ": " << norm << endl;
                }

                // Z-SCORE NORMALIZATION
                cout << "Z-score Normalization" << endl;

                // step 1-- calc average salary
                double total = 0;

                for (int i = 0; i < numEmployees; i++) 
                    total += salary[i]; // 1st step getting average -> first add up all the salaries

                double average = total / numEmployees; //^ calculate average by dividing the total salary by number of employees

                // step 2--deleting avg from each salary (x-avg)
                vector<double> zscore(numEmployees); // vector to store zscore normalized values
                
                for (int i = 0; i < numEmployees; i++) // loop through each employee and calculate each deviation
                    zscore[i] = salary[i] - average; //^

                //step 3-- add up squared differences
                double squaredTotal = 0;
                
                for (int i = 0; i < numEmployees; i++) // loop through each employee and add the squared values.
                    squaredTotal += zscore[i] * zscore[i];

                double variance = squaredTotal / numEmployees;
                
                //step 4--square root
                double stdDev = sqrt(variance); //square root the variance

                //step 5-- final step for z-score
                for (int i = 0; i < numEmployees; i++) { //finally loop through each employee and divide by the stdDev
                    zscore[i] = zscore[i] / stdDev;
                    cout << "Employee " << employeeName[i] << ": " << zscore[i] << endl;
                }

                return 0;
            }

Key features:

This C++ program performs two fundamental data normalization techniques—Min-Max Normalization and Z-Score Normalization—on a list of employee salaries provided by the user. It begins by collecting employee names and corresponding salary values, then calculates:

  • Min-Max Normalization, which scales salaries to a range between 0 and 1 based on the minimum and maximum salaries in the dataset.

  • Z-Score Normalization, which standardizes salaries by centering them around the mean and scaling by the standard deviation.

These normalization methods are key in data mining and machine learning preprocessing to ensure fair comparisons between numerical values and prepare datasets for further analysis. The program emphasizes practical usage of vectors, loops, conditionals, and statistical formulas in C++.

                        bool IsSorted(int numArray[], int numArrayElems) // state whether the inputs are already sorted or not
                        {
                            for (int i=1; i<numArrayElems; i++)
                            {
                                if(numArray[i]<numArray[i-1])
                                {
                                    return false;
                                }
                            }
                            return true;
                        }
                        
                            
                        int reSwap(int numArray[], int newValue, int numArrayElems) // used if list is already sorted to find where to swap and add the new input
                        {
                            for(int i=0; i<numArrayElems; i++)
                            {
                                if(numArray[i] >= newValue)
                                {
                                    return (i);
                                }
                            }
                            return numArrayElems;
                        }

                        void insertInto(int numArray[], int& numArrayElems, bool IsSorted) // function to ask for new value and place it according to whether the list is already sorted or not
                        {
                            int newValue, re;
                            cout<< "Enter a value to insert: ";
                            cin>> newValue;
                            
                            if(IsSorted) // if already sorted call for reSwap function to find the correct spot to place the new value
                            {
                                re= reSwap(numArray, newValue, numArrayElems);
                                for(int i=numArrayElems; i>re; i--)
                                {
                                    numArray[i] = numArray[i-1];
                                }
                                numArray[re] = newValue;
                            }
                            else {
                                numArray[numArrayElems] = newValue;
                            }
                            numArrayElems++;
                        }
                        
                        int lookinArray(int numArray[], int numArrayElems, int delValue) // function to check if value to delete is in the actual function
                        {
                            for(int i=0; i<numArrayElems; i++) //is found
                                if(numArray[i] == delValue)
                                    return(i);
                        
                            return(-1);// if value to delete is not found in list then -1 is returned= 'not found'

                        void selectionSort(int numArray[], int numArrayElems) //sorting list from small to big
                        {
                            bool swap;
                            do{
                                swap=false;
                                for (int i=0; i<numArrayElems-1;i++)
                                {
                                    if (numArray[i]>numArray[i+1])
                                    {
                                        float temp=numArray[i]; //swap elements
                                        numArray[i]=numArray[i+1];
                                        numArray[i+1]=temp;
                                        swap=true;
                                    }
                                }
                            } while(swap);
                            return;
                        }
                        
                        void reverse(int numArray[], int numArrayElems) // reversing the list
                        {
                            for(int i=0; i<numArrayElems/2; i++) //swap from the middle of the array
                            {
                                int temp;
                                temp=numArray[i];
                                numArray[i]=numArray[numArrayElems-i-1];
                                numArray[numArrayElems-i-1]=temp; //same idea to sorting
                            }
                        
                        }

                            }

                            reverse(numArray, numArrayElems);

C++ to MIPS Conversion

This MIPS assembly program mirrors a C++ routine that processes an integer array by multiplying a running total by 8 and then adding each array element. It showcases how high-level control flow and arithmetic operations are implemented in a low-level language using register manipulation and memory addressing.

Technolgy used: MIPS Assembly Language

                        #RAT
                            # $s0 MyArray
                            # $s1 total
                            # $s2 index
                            # $s3 MAX
                            # $s4 MyArray[index]

                        .data

                            message: .asciiz "Final total = "
                        
                        .text
                            la $s0, myArray #load address myArray --> base address
                            li $s1, 0 #initialize total = 0 
                            li $s2, 0 #initialize index = 0 
                            lw $s3, MAX #load max value ->6 into $s3

                                #total*=8--> total=total*8

                                add $t1, $s0, $t0 #add base address of myarray and offset 
                                lw $s4, ($t1) #arrays total location at $t1 
                                
                                #total=total + MyArray[index];
                                add $s1, $s1, $s4

                            endwhile:

                                syscall
